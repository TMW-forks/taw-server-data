907-1.gat,x,y,0	script	Kaaussa	000, {



	mes "[Kaaussa]";
	mes "Hi. What do you want?";
	menu
		"Can you make nettle shirt for me?", -,
		"Can you make protecting chaps for me?", L_Pr,
		"Nothing", L_No;

	mes "[Kaaussa]";
	mes "To make nettle shirt I need 2 nettle cloth and nettle string. Have you got them?";
	menu
		"Yes", -,
		"No", L_No;

	if (countitem("nettlecloth") < 2) goto L_NotEnought;
	if (countitem("nettlestring") < 1) goto L_NotEnought;
	mes "[Kaaussa]";
	mes "Here you are.";
	delitem "nettlestring", 1;
	delitem "nettlecloth", 2;
	getitem "nettleshirt", 1;
	next;
	close;

L_Pr:
	mes "[Kaaussa]";
	mes "To make protecting chaps I need 1 boar skin, 2 nettle cloth, 2 nettle string. Have you got them?";
	menu
		"Yes", -,
		"No", L_No;

	if (countitem("boarskin") < 1) goto L_NotEnought;
	if (countitem("nettlecloth") < 2) goto L_NotEnought;
	if (countitem("nettlestring") < 2) goto L_NotEnought;
	mes "[Kaaussa]";
	mes "Here you are.";
	delitem "boarskin", 1;
	delitem "nettlecloth", 2;
	delitem "nettlestring", 2;
	getitem "protectingchaps", 1;
	next;
	close;

L_No:
	close;

L_NotEnought:
	mes "[Kaaussa]";
	mes "It looks like you haven't.";
	close;
}
